        let currentIdx = 0;
        let scoreA = 0;
        let scoreB = 0;

        function startQuiz() {
            document.getElementById('start-screen').classList.add('hidden');
            document.getElementById('quiz-screen').classList.remove('hidden');
            showQuestion();
        }

        function showQuestion() {
            document.getElementById('progress').innerText = (currentIdx + 1) + " / " + questions.length;
            document.getElementById('question-text').innerText = questions[currentIdx].q;
            document.getElementById('optA').innerText = "A: " + questions[currentIdx].a;
            document.getElementById('optB').innerText = "B: " + questions[currentIdx].b;
        }

        function nextQuestion(choice) {
            if(choice === 'A') scoreA++;
            else scoreB++;

            currentIdx++;
            if(currentIdx < questions.length) {
                showQuestion();
            } else {
                showResult();
            }
        }

        function showResult() {
            document.getElementById('quiz-screen').classList.add('hidden');
            document.getElementById('result-screen').classList.remove('hidden');

            let type = "";
            let desc = "";

            if (scoreA >= 11) {
                type = "【忠実な愛されキャラ！犬系タイプ】";
                desc = "あなたは感情表現が豊かで誠実な人。でも寂しさが極まると「構ってちゃん」が加速し、しっぽを振りすぎて相手を疲れさせてしまうかも…？";
            } else if (scoreB >= 11) {
                type = "【自由奔放なツンデレ！猫系タイプ】";
                desc = "ミステリアスな魅力を持つあなた。独占欲は人一倍で、一度「自分のもの」と決めた相手への執着は、誰にも止められない禁断レベルかも。";
            } else {
                type = "【ハイブリッド！うさぎ系タイプ】";
                desc = "人懐っこさと繊細さを併せ持つ、孤独に弱いタイプ。放置されると精神的に枯れてしまうので、常に愛情確認を求めてしまう傾向があります。";
            }

            document.getElementById('res-type').innerText = type;
            document.getElementById('res-desc').innerText = desc;
        }

        function shareTwitter() {
            const result = document.getElementById('res-type').innerText;
            const text = encodeURIComponent("恋愛診断「犬系か猫系か」の結果は..." + result + "\\n#犬猫恋愛診断 #心理テスト");
            const url = encodeURIComponent(window.location.href);
            window.open(`https://twitter.com/intent/tweet?text=${text}&url=${url}`, '_blank');
